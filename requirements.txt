# Mobius Chat — runtime dependencies (2026-04-20 clean-slate rewrite).
#
# Audit policy
# ------------
# * One line = one package, lowest version that works in prod. Ranges
#   use ``>=`` to allow patch-level security fixes; upper bounds only
#   when a breaking change is known.
# * Dev/test deps (pytest, coverage, mypy) are NOT here — they live in
#   a future requirements-dev.txt. The Cloud Run image should never
#   carry a test runner.
# * Sibling editable installs (mobius-contracts, mobius-skills-core,
#   mobius-retriever) are handled by the Dockerfile's multi-stage
#   COPY+install, NOT by references here. Keeps ``pip install -r
#   requirements.txt`` usable in raw dev checkouts without weird
#   relative paths.
# * When you add a dep: bump the Dockerfile builder cache by
#   touching this file, not by rebuilding from scratch.

# ── Web stack ────────────────────────────────────────────────────────
fastapi>=0.104.0              # app, dependency injection, lifespan
uvicorn[standard]>=0.24.0     # ASGI server + uvloop + httptools
pydantic>=2.0.0               # request/response schemas
python-dotenv>=1.0.0          # dev-only .env loading (no-op in hosted)
# Required by /chat/roster-upload (File/Form params). Historically pulled
# in as a transitive dep of something else, but that resolution is brittle —
# pinning explicitly so the next dep-graph shift doesn't break startup.
python-multipart>=0.0.6

# ── Structured logs (Sprint 1 #10, 2026-04-23) ───────────────────────
# python-json-logger emits one JSON object per log record so Cloud
# Logging parses severity + jsonPayload fields natively (rather than
# forcing us to regex text in Log Explorer). The formatter is opt-in
# via env — dev defaults to human-readable, hosted flips to JSON.
python-json-logger>=2.0.7

# ── Distributed tracing (Sprint 1 #11, 2026-04-24) ───────────────────
# OpenTelemetry API + SDK + the GCP Cloud Trace exporter. Auto-
# instrumentation packages add spans for FastAPI incoming requests
# and httpx outgoing calls without any code change; we only add
# manual spans at pipeline + LLM boundaries.
#
# Tracing is env-gated (CHAT_TRACE_ENABLED=0 by default in dev, on
# in hosted envs via deploy.env). Initialization is lazy: when the
# gate is off, the OTel SDK isn't even imported so dev boot stays
# fast and the dep is effectively free.
opentelemetry-api>=1.27.0
opentelemetry-sdk>=1.27.0
opentelemetry-exporter-gcp-trace>=1.7.0
opentelemetry-instrumentation-fastapi>=0.48b0
opentelemetry-instrumentation-httpx>=0.48b0

# ── HTTP ─────────────────────────────────────────────────────────────
httpx>=0.25.0                 # outbound HTTP (skills-mcp, task-manager, doc-reader)

# ── Auth ─────────────────────────────────────────────────────────────
PyJWT>=2.8.0                  # HS256 token validation (shared with mobius-os)

# ── GCP SDK ──────────────────────────────────────────────────────────
# ``google-cloud-aiplatform`` brings ``vertexai`` (Gemini + embeddings)
# and pulls ``google-auth``, ``google-api-core``, grpcio as transitive.
# ``google-cloud-secret-manager`` is required for the secrets_loader
# hosted-env path; lazy-imported so dev without the dep still works.
google-cloud-aiplatform>=1.38.0
google-cloud-secret-manager>=2.20.0

# ── Database ─────────────────────────────────────────────────────────
# psycopg2-binary: sync driver used by ~33 call sites for Postgres.
#   ``binary`` variant avoids needing libpq-dev at build time (we get
#   it from apt as libpq5 at runtime).
# asyncpg: async driver for analytics/telemetry writes that would block
#   the event loop with psycopg2.
psycopg2-binary>=2.9.9
asyncpg>=0.29.0

# ── Retrieval / vector ───────────────────────────────────────────────
chromadb>=0.4.0               # local vector store (thread-corpus, instant-RAG)

# ── LLM output repair ────────────────────────────────────────────────
# json-repair catches the common LLM JSON failures (trailing commas,
# truncated objects, single-quote keys) before handing off to a
# slower LLM repair round. Saves ~400 ms on integrator retries.
json-repair>=0.55.0

# ── Config ───────────────────────────────────────────────────────────
PyYAML>=6.0                   # config/payer_normalization.yaml etc.

# ── Document parsing ─────────────────────────────────────────────────
# PyMuPDF (``fitz``) — PDF extraction for uploaded documents
# (doc_reader proxy, roster uploads). Pinned under the major version
# to avoid the parser rewrites that shipped in 2.x.
PyMuPDF>=1.24.0,<2.0.0

# ── Web scraping ─────────────────────────────────────────────────────
beautifulsoup4>=4.12.0        # bs4 import; powers skills-core.web_scrape

# ── Search fallback ──────────────────────────────────────────────────
# Used by skills-core.google_search when GOOGLE_SEARCH_URL isn't
# pointed at the microservice and GOOGLE_CSE credentials aren't set.
# No API key required; honors robots.txt; low quality but works.
# When the microservice deploys, this stays installed but never fires.
ddgs>=9.0.0

# ── Optional: Redis queue ────────────────────────────────────────────
# Monolith beta uses CHAT_QUEUE_TYPE=memory; this dep is included so
# switching to redis is a config-only change with no rebuild.
redis>=5.0.0

# ── Optional: MCP client ─────────────────────────────────────────────
# Only used when CHAT_SKILLS_MCP_URL is set AND skills are routed
# through MCP rather than being invoked as in-process Python imports.
# Beta monolith uses the in-process path so this is currently
# unused-at-runtime but kept for the post-beta split.
mcp>=1.0.0
